Regular Expressions

POSIX extended regular expressions in Elysium. Intermediate

Available Functions

func main() {
    // Test — does the pattern match anywhere?
    let hasDigits = regex.test("\\d+", "hello 123 world")
    // hasDigits = true

    // Match — extract the first match
    let matched = regex.match("\\d+", "abc 456 def")
    // matched = "456"

    // Search — find starting position of first match
    let pos = regex.search("\\d+", "abc 789 def")
    // pos = 4

    // Replace — replace first occurrence
    let replaced = regex.replace("foo", "hello foo world", "bar")
    // replaced = "hello bar world"

    // Split — split string by pattern
    let parts = regex.split(",", "a,b,c")
    // parts = "a,b,c" (comma-joined for now)
}
