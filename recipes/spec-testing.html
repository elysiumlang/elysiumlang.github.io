Spec-Driven Testing

Write inline tests with spec/feat/expect. Beginner

Basic Spec

/// Summary: Example demonstrating spec-driven development.
func add(a: Int, b: Int) -> Int {
    a + b
}

spec "Calculator" {
    feat "adds two numbers" {
        let result = add(2, 3)
        expect result == 5
    }
    feat "adds negative numbers" {
        let result = add(-1, -2)
        expect result == -3
    }
}

Using describe / it Synonyms

describe "Greeter" {
    it "says hello" {
        let msg = greet("World")
        expect msg == "Hello, World"
    }
}

Using todo / question Markers

spec "Calculator" {
    feat "adds two numbers" {
        let result = add(2, 3)
        expect result == 5
        todo "test negative numbers"
    }
    feat "division" {
        todo
        question "what about division by zero?"
    }
}

Specs with Imports

import "./math.ely" as math

spec "Math" {
    feat "square works" {
        let result = math.square(4)
        expect result == 16
    }
    feat "double works" {
        let result = math.double(3)
        expect result == 6
    }
}

Key Points

  • spec / describe are interchangeable synonyms.
  • feat / it are interchangeable synonyms.
  • expect <expr> asserts that the expression is truthy.
  • todo and question compile to nil — they are zero-cost documentation markers.
  • Specs are first-class language constructs, not a separate framework.
