@elysium/langgraph 0.1.0

Elysium port of LangGraph by LangChain Inc — stateful graph-based agent orchestration.

epm install @elysium/langgraph

Import

import "#/@elysium/langgraph" as langgraph

API

createRegistry() -> StringCreate a new graph registry
graph(registry, description) -> StringCreate a new state graph
addNode(registry, gid, name, nodeSpec) -> StringAdd a processing node
addEdge(registry, gid, from, to) -> StringAdd a directed edge
addConditionalEdges(registry, gid, from, condition, mapping) -> StringAdd conditional routing
compile(registry, gid) -> StringValidate graph structure
invoke(registry, gid, initial) -> StringRun the graph with initial state
stream(registry, gid, initial) -> StringExecute and stream intermediate states
getState(registry, gid) -> StringGet current state snapshot
updateState(registry, gid, val) -> StringUpdate the graph state
branch(registry, gid, branchSpec) -> StringAdd parallel branching
interrupt(registry, gid, val) -> StringPause the graph (human-in-the-loop)
resume(registry, gid) -> StringResume the graph from an interrupt
